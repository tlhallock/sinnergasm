#!/bin/bash

rm -f *.crt *.srl *.key *.pem *.csr

openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout ca.key -out ca.crt -config ca_req.conf
openssl genpkey -algorithm RSA -out server.key
openssl req -new -key server.key -out server.csr -config server_req.conf
openssl x509 -req -in server.csr -CA ca.crt -CAkey ca.key -CAcreateserial -out server.crt -days 365 -extfile v3_end.cnf -extensions v3_end

cat server.crt server.key > server.pem
